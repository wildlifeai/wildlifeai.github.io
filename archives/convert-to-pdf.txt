name: Generate PDFs from Jekyll HTML

on:
  push:
    branches: [main] # Adjust if your main branch is named differently

jobs:
  generate-pdfs:
    runs-on: ubuntu-latest

    steps:
      - name: ğŸ“¥ Checkout repository
        uses: actions/checkout@v3

      - name: ğŸ’ Set up Ruby
        uses: ruby/setup-ruby@v1
        with:
          ruby-version: '3.2'

      - name: ğŸ”§ Install Jekyll
        run: gem install jekyll

      - name: ğŸ› ï¸ Build Jekyll site
        run: jekyll build

      - name: ğŸ“¦ Install wkhtmltopdf
        run: |
          sudo apt-get update
          sudo apt-get install -y wkhtmltopdf

      - name: ğŸ§¹ Clean old PDFs
        run: |
          mkdir -p docs/_pages/pdf
          rm -f docs/_pages/pdf/*.pdf

      - name: ğŸ“„ Convert built HTML to PDFs
        run: |
          ls -a -R _site/
          mkdir -p docs/_pages/pdf
          echo "ğŸ” Searching for .html files under _site/_pages/md"
          find _site/_pages/md -name '*.html' | while read -r html; do
            filename=$(basename "$html" .html)
            output="docs/_pages/pdf/${filename}.pdf"
            echo "ğŸ“¤ Converting $html â†’ $output"
            wkhtmltopdf --enable-local-file-access "$html" "$output" \
              && echo "âœ… Created $output" || echo "âŒ Failed to create $output"
          done
          echo "ğŸ“ Output folder contents:"
          ls -lh docs/_pages/pdf



      # - name: ğŸ§ª wkhtmltopdf sanity test

      #     wkhtmltopdf test.html docs/pages/pdf/test.pdf
      #     ls -lh docs/pages/pdf

      - name: ğŸ§¾ Git status before commit
        run: |
          git status
          find docs/_pages/pdf -type f

      - name: ğŸš€ Commit and push generated PDFs
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add docs/_pages/pdf/*.pdf || true
          git status
          git commit -m "Update PDFs from built HTML [skip ci]" || echo "ğŸŸ¡ No changes to commit"
          git push origin main || echo "ğŸŸ  Push skipped (maybe no new PDFs)"
